### Официальная документация React i18next

Основной ресурс, который вам нужен — это **[react.i18next.com](https://react.i18next.com/)**. Это отдельный сайт-сателлит. Там есть раздел **"Step by step guide"**, который является «золотым стандартом» для старта. В отличие от базового i18next, здесь объясняется специфика React: хуки (`useTranslation`), компоненты-обертки и работа с Context API.


i18n.ts
---
```
import i18n from "i18next";
import { initReactI18next } from "react-i18next";
import LanguageDetector from "i18next-browser-languagedetector";

const resources = {
  en: {
    translation: {
      "Calories tracker": "Calories tracker",
      "Total calories:": "Total calories:",
      "Not Found Page...": "Not Found Page...",
      "Add new Meal": "Add new Meal",
    },
  },
  ru: {
    translation: {
      "Calories tracker": "Подсчет калорий",
      "Total calories:": "Количество калорий:",
      "Not Found Page...": "Страница не найдена",
      "Add new Meal": "Добавить блюдо"
    },
  },
};

  
void i18n
  .use(LanguageDetector) // Автоопределение языка
  .use(initReactI18next) // Передача i18n в react-i18next
  .init({
    resources,
    fallbackLng: "en", // Язык по умолчанию
    interpolation: {
      escapeValue: false, // React сам защищает от XSS
    },
  });

export default i18n;
```

example: Header.tsx

```
import { useTranslation } from "react-i18next";

const Header = () => {
  const { t, i18n } = useTranslation();
  const changeLanguage = (lng: string) => {
    i18n.changeLanguage(lng);
  };
  
}

return (
          <Typography
            onClick={() => navigate("/")}
            variant="h6"
            component="div"
            sx={{ flexGrow: 1, cursor: "pointer" }}
          >
            {t("Calories tracker")}
          </Typography>
  
          <Box>
            <Button
              variant="text"
              sx={{ color: "white" }}
              onClick={() => changeLanguage("ru")}
            >
              ru
            </Button>
            <Button
              variant="text"
              sx={{ color: "white" }}
              onClick={() => changeLanguage("en")}
            >
              en
            </Button>
)
```
